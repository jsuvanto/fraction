124 void Fraction::simplify() {
125  int a = p_;
126  int b = q_;
127  // Euclidean method for finding the greatest common divisor of a and b.
128  while (b != 0) {
129  int tmp = a;
130  a = b;
131  b = tmp % a;
132  }
133  p_ = p_/a;
134  q_ = q_/a;
135  // Change the sign from the denominator to the numerator.
136  // This relies on the fact that if q_ is negative, p_ isn't.
137  if (q_ < 0) {
138  p_ = -p_;
139  q_ = -q_;
140  }
141 }
